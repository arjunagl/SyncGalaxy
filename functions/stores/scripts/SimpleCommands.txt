
--------------Deploying to aws, this works
//Create the bucket
aws s3 mb s3://SyncGalaxy-bucket

//Create the package
aws cloudformation package --template-file build/storesSAM.yaml --output-template-file build/storesServerless-output.yaml --s3-bucket SyncGalaxy-bucket

//Deploying the pakcage
aws cloudformation deploy --template-file build/storesServerless-output.yaml --stack-name SyncGalaxy-Stack --capabilities CAPABILITY_IAM

//Inserting sample data
 aws dynamodb batch-write-item --request-items file://scripts/sampleData.json

//To get the list of Errors
aws cloudformation describe-stack-events --stack-name SyncGalaxy-Stack
