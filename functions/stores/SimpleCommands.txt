Deploy
aws cloudformation deploy --template-file .\cloudFormation.yaml --stack-name Sync-Galaxy-Stack  --capabilities CAPABILITY_IAM

Update
aws cloudformation update-stack --stack-name Sync-Galaxy-Stack --template-body file://cloudFormation.yaml --capabilities CAPABILITY_IAM

Delete
aws cloudformation delete-stack --stack-name Sync-Galaxy-Stack

Fetch Errors
aws cloudformation describe-stack-events --stack-name Sync-Galaxy-Stack
